   15 -- The type constructor PD is defined in Types
   16 instance Monad PD where
   17   return x = PD [(1, x)]
   18   (PD xs) >>= f = PD $ concatMap (\(p, x) -> scaleProb p (runPD (f x))) xs
   19     where
   20       scaleProb p ys = [(p * q, y) | (q, y) <- ys]

   29 -- | `uniform` generates a uniform probability distribution for the integers from 0 to n-1.
   30 -- If n is zero or negative, it will produce an error as a uniform distribution is not defined for these cases.
   31 uniform :: Int -> PD Int
   32 uniform n
   33   | n > 0 = PD [(1 % fromIntegral n, x) | x <- [1..n]]
   34   | otherwise = error "uniform: non-positive argument"

   37 normalize :: Ord a => PD a -> PD a
   38 normalize (PD xs) = PD $ map normalizeProb $ accumulate xs
   39   where
   40     -- accumulate goes through each (probability, outcome) pair and accumulates probabilities for each unique outcome
   41     accumulate :: Ord a => [(Prob, a)] -> [(Prob, a)]
   42     accumulate [] = []
   43     accumulate ((p, x):ys) = (totalP, x) : accumulate (filter ((/= x) . snd) ys)
   44       where
   45         totalP = p + sum [pi | (pi, xi) <- ys, xi == x]
   46 
   47     -- normalizeProb adjusts each probability to ensure the total is 1
   48     normalizeProb :: (Prob, a) -> (Prob, a)
   49     normalizeProb (p, x) = (p / totalProb, x)
   50 
   51     -- Calculate the total probability of the original list
   52     totalProb = sum $ map fst xs
   53 
   54 expectation :: Fractional a => PD (Maybe a) -> (Prob, a)
   55 expectation (PD xs) = if successProb == 0
   56                       then error "Expected value is undefined"
   57                       else (successProb, weightedAvg)
   58   where
   59     justVals = [(p, x) | (p, Just x) <- xs]
   60     successProb = sum $ map fst justVals
   61     weightedAvg = sum [fromRational p * x | (p, x) <- justVals] / fromRational successProb

   64 -- The type Bag is defined in Types
   65 union :: Bag -> Bag -> Bag
   66 union (Bag xs) (Bag ys) = Bag (merge xs ys)
   67   where
   68     merge [] bs = bs
   69     merge as [] = as
   70     merge (a:as) (b:bs)
   71       | a < b     = a : merge as (b:bs)
   72       | otherwise = b : merge (a:as) bs
   73 
   74 type Value = Bag  -- all elements must be >=0
   75 
   76 type Store = [(VName,Value)] -- all vnames must be distinct
   77 
   78 newtype Calc a = Calc {runCalc :: Store -> PD (Either RunError (a,Store))}
   79 
   80 instance Monad Calc where
   81   return x = Calc $ \s -> PD [(1, Right (x, s))]
   82   (Calc c) >>= f = Calc $ \s -> PD $ do
   83       (prob, result) <- runPD (c s)
   84       case result of
   85           Left err -> [(prob, Left err)]
   86           Right (a, s') -> runPD $ runCalc (f a) s'
   87 
   88 -- You shouldn't need to modify these
   89 instance Applicative Calc where
   90   pure = return; (<*>) = ap
   91 instance Functor Calc where
   92   fmap = liftM

   94 fault :: RunError -> Calc a
   95 fault re = Calc $ \_ -> PD [(1, Left re)]
   96 
   97 lookVar :: VName -> Calc (Maybe Value)
   98 lookVar x = Calc $ \s -> PD [(1, Right (lookup x s, s))]
   99 
  100 setVar :: VName -> Maybe Value -> Calc ()
  101 setVar x mv = Calc $ \s -> PD [(1, Right ((), updateStore x mv s))]
  102   where
  103     updateStore :: VName -> Maybe Value -> Store -> Store
  104     updateStore varName maybeVal store =
  105       case maybeVal of
  106         Just val -> (varName, val) : filter ((varName /=) . fst) store
  107         Nothing  -> filter ((varName /=) . fst) store
  108 
  109 pick :: PD a -> Calc a
  110 pick pd = Calc $ \s -> PD $ concatMap (\(prob, val) -> [(prob, Right (val, s))]) (runPD pd)
  111 
  112 eval :: Exp -> Calc Value
  113 eval (Cst i)       = return $ Bag [i] -- Constant integers are represented as singleton bags.
  114 eval (Var v)       = do -- Look up a variable and return its value.
  115   maybeValue <- lookVar v
  116   case maybeValue of
  117     Just value -> return value
  118     Nothing -> fault $ Unbound v "Variable not found"
  119 eval (Join exps)   = do -- Evaluate each expression and join their results.
  120   bags <- mapM eval exps
  121   return $ foldr union (Bag []) bags
  122 eval (Let v e1 me2) = do -- Evaluate e1, bind its value to v, then evaluate me2 if present.
  123   val1 <- eval e1
  124   setVar v (Just val1)
  125   case me2 of
  126     Just e2 -> eval e2
  127     Nothing -> return val1
  128 eval (Seq e1 e2) = eval e1 >> eval e2
  129 eval (Is e1 e2) = do
  130   val1 <- eval e1
  131   val2 <- eval e2
  132   return $ if val1 == val2 then Bag [1] else Bag []
  133 eval (Sum e) = do
  134   Bag ints <- eval e
  135   return $ Bag [sum ints]
  136 eval (Count e) = do
  137   Bag ints <- eval e
  138   return $ Bag [length ints]
  139 eval (Times e1 e2) = do
  140   Bag ints1 <- eval e1
  141   Bag ints2 <- eval e2
  142   return $ Bag [x * y | x <- ints1, y <- ints2]
  143 eval (Roll e) = do
  144   Bag ints <- eval e
  145   if null ints then fault $ EmptyChoice "Cannot roll an empty bag"
  146   else do
  147     let n = head ints
  148     if n <= 0 then
  149         fault $ OtherError "Cannot roll a die with non-positive sides"
  150     else do
  151         rolledNumber <- pick $ uniform n
  152         return $ Bag [rolledNumber]
  153 eval (Take sel v) = do
  154   maybeVal <- lookVar v
  155   case maybeVal of
  156     Just (Bag ints) -> case sel of
  157       Min -> return . Bag $ [minimum ints | not (null ints)]
  158       Max -> return . Bag $ [maximum ints | not (null ints)]
  159       Rand -> if null ints then fault $ EmptyChoice "Cannot pick from an empty bag"
  160               else pick $ uniform (length ints) >>= \i -> return $ Bag [ints !! i]
  161     Nothing -> fault $ Unbound v "Variable not found in Take"
  162

  164 evaluate :: Exp -> PD (Either RunError Value)
  165 evaluate exp = normalize $ PD $ concatMap processOutcome (runPD outcomesPD)
  166   where
  167     initialStore = [] :: Store
  168     outcomesPD = runCalc (eval exp) initialStore
  169     processOutcome :: (Prob, Either RunError (Value, Store)) -> [(Prob, Either RunError Value)]
  170     processOutcome (prob, Right (val, _)) = [(prob, Right val)]
  171     processOutcome (prob, Left err) = [(prob, Left err)]
