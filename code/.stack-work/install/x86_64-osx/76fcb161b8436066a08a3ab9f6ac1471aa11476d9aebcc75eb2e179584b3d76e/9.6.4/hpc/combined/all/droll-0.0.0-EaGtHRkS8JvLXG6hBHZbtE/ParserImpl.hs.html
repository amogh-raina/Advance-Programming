   11 parseExp :: String -> Either String Exp
   12 parseExp input = case parse (expParser <* eof) "" input of
   13     Left err -> Left $ show err
   14     Right expr -> Right expr

   16 -- Main parser for expressions (Exp)
   17 expParser :: Parser T.Exp
   18 expParser = rollParser <|>  letParser <|>  additionParser <|> try sExpParser
   19 
   20 letParser :: Parser T.Exp
   21 letParser = do
   22   spaces
   23   string "let"
   24   spaces
   25   varName <- vNameParser
   26   spaces
   27   string "be"
   28   spaces
   29   exp1 <- sExpParser
   30   spaces
   31   string "in"
   32   spaces
   33   notFollowedBy (string "") -- Ensures that the next parser doesn't match the empty string
   34   T.Let varName exp1 . Just <$> expParser
   35 
   36 letInExpParser :: String -> T.Exp -> Parser T.Exp
   37 letInExpParser varName exp1 = do
   38   string "in"
   39   spaces
   40   T.Let varName exp1 . Just <$> expParser
   41 
   42 vNameParser :: Parser String
   43 vNameParser = many1 letter
   44 
   45 rollParser :: Parser T.Exp
   46 rollParser = do
   47     string "roll"
   48     spaces
   49     expr <- try (exprInParentheses <* spaces) <|> sExpParser
   50     return $ T.Roll expr
   51 
   52 -- Define a separate parser for expressions within parentheses
   53 exprInParentheses :: Parser T.Exp
   54 exprInParentheses = between (char '(' *> spaces) (spaces <* char ')') expParser
   55 
   56 -- Parser for simple expressions (SExp)
   57 sExpParser :: Parser T.Exp
   58 sExpParser = try additionParser <|> basicSExpParser
   59 
   60 -- Addition operator '+'
   61 additionParser :: Parser T.Exp
   62 additionParser = comparisonLevelParser `chainl1` addOp
   63 
   64 -- Addition operation parser
   65 addOp :: Parser (T.Exp -> T.Exp -> T.Exp)
   66 addOp = do
   67     spaces
   68     char '+'
   69     spaces
   70     return (\exp1 exp2 -> T.Sum (T.Join [exp1, exp2]))
   71 
   72 basicSExpParser :: Parser T.Exp
   73 basicSExpParser =  numParser
   74                <|>  takeParser
   75                <|>  countParser
   76                <|>  sumParser
   77                <|>  varParser
   78                <|>  joinParser
   79                <|>  timesParser
   80                <|>  comparisonParser
   81 
   82 
   83 numParser :: Parser T.Exp
   84 numParser = do
   85   spaces
   86   num <- many1 digit
   87   spaces
   88   return $ T.Cst (read num)
   89 
   90 varParser :: Parser T.Exp
   91 varParser = try $ do
   92   spaces
   93   var <- many1 letter
   94   spaces
   95   return $ T.Var var
   96 
   97 sumParser :: Parser T.Exp
   98 sumParser = do
   99   spaces
  100   string "sum"
  101   spaces
  102   T.Sum <$> additionWithoutSumParser
  103 
  104 -- New parser for handling addition without wrapping in Sum
  105 additionWithoutSumParser :: Parser T.Exp
  106 additionWithoutSumParser = comparisonLevelParser `chainl1` addOpWithoutSum
  107 
  108 -- Modified addition operation parser without Sum
  109 addOpWithoutSum :: Parser (T.Exp -> T.Exp -> T.Exp)
  110 addOpWithoutSum = do
  111     spaces
  112     char '+'
  113     spaces
  114     return (\exp1 exp2 -> T.Join [exp1, exp2])
  115 
  116 countParser :: Parser T.Exp
  117 countParser = do
  118   spaces
  119   string "count"
  120   spaces
  121   T.Count <$> sExpParser
  122 
  123 -- 'times' operator (right associative)
  124 timesParser :: Parser T.Exp
  125 timesParser = do
  126   exp1 <- basicSExpParser
  127   spaces
  128   string "times"
  129   spaces
  130   exp2 <- expParser -- Right associativity: allow timesParser to be nested on the right
  131   return $ T.Times exp1 exp2
  132 
  133 -- Intermediate precedence level for 'times'
  134 timesLevelParser :: Parser T.Exp
  135 timesLevelParser = try timesParser <|> basicSExpParser
  136 
  137 takeParser :: Parser T.Exp
  138 takeParser = do
  139   try (string "take") -- Use 'try' for backtracking if needed
  140   spaces
  141   sel <- (string "highest" >> return T.Max) <|> (string "lowest" >> return T.Min) <|> (string "any" >> return T.Rand)
  142   spaces
  143   string "from"
  144   spaces
  145   case sel of
  146     T.Max -> T.Take Max <$> vNameParser
  147     _ -> T.Take sel <$> numParserC
  148 
  149 numParserC :: Parser String
  150 numParserC = do
  151   spaces
  152   num <- many1 digit
  153   spaces
  154   return num
  155 
  156 -- Comparison operators ('is', 'is not')
  157 comparisonParser :: Parser T.Exp
  158 comparisonParser = do
  159   exp1 <- timesLevelParser
  160   spaces
  161   _ <- try (string "is not") <|> string "is"
  162   spaces
  163   Is exp1 <$> timesLevelParser
  164 
  165 comparisonLevelParser :: Parser T.Exp
  166 comparisonLevelParser = try comparisonParser <|> timesLevelParser
  167 
  168 joinParser :: Parser T.Exp
  169 joinParser = do
  170     char '('
  171     exps <- sepBy expParser (spaces >> char ',' >> spaces)
  172     char ')'
  173     return $ flattenJoins exps
  174 
  175 flattenJoins :: [T.Exp] -> T.Exp
  176 flattenJoins exps = let flattened = concatMap (\e -> case e of
  177                                                        T.Join es -> es
  178                                                        _ -> [e])
  179                                               exps
  180                     in case flattened of
  181                          [e] -> e  -- Avoid single-expression Joins
  182                          _ -> T.Join flattened
